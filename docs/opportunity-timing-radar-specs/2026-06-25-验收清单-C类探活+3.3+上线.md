### B1 live 集成脚本（可独立重跑核对，直连真源、零库写）
```python
# 保存为 /tmp/verify_cclass.py，repo 的 crawler 目录下跑：
#   cd <repo>/crawler && python3 /tmp/verify_cclass.py
# 沙箱可能需 bypass 联网。下方 id 为 2026-06-25 实采（在招岗会随时间撤下，DEAD 项恒稳）。
import sys; sys.path.insert(0,".")

def run(a,u,expect):

        enrich.ENRICH_REGISTRY[a]({"jd_url":u},{}); got="ALIVE"
    except enrich.JobClosedError: got="DEAD"
    except Exception as e: got=f"MISS/{type(e).__name__}"  # 网络错=拿不准，重试
    print(f"[{'PASS' if expect in got else 'XXXX'}] {a:11} expect={expect:5} got={got}")
cases=[
 ("tencent","https://careers.tencent.com/jobdesc.html?postId=2049458033221873664","DEAD"),
 ("apple","https://jobs.apple.com/en-us/details/200667252-0836/x?team=SFTWR","DEAD"),

 ("sf_express","https://hr.sf-express.com/JobSearchById/99999999,3","DEAD"),
 ("amazon","https://www.amazon.jobs/en/jobs/99999999/x","DEAD"),
 # 在招样本（可能随时间撤下→届时换一个当前 active 的）：
```
